public static int totalOccurencesOfCharacter(String s, char ch){   
	  char[] S = s.trim().toCharArray();
	  Arrays.sort(S);
	  s = new String(S);
	  int first = s.indexOf(ch);
	  int last = s.lastIndexOf(ch);
	  System.out.println(first+" "+last);
	  return (last == first) ? 1 : (last-first+1);
	}
