class Solution {
    public int maxProfit(int[] prices) 
    { 
     int max_profit = 0, min = Integer.MAX_VALUE, c = 0;
     for(int i = 0; i < prices.length; i++)
     {
         if(prices[i] < min)
         {
             min = prices[i];
         }
         else
         {
           c = prices[i] - min;
           if(c > max_profit)
           {
               max_profit = c;
           }
         }
     }
        return max_profit;
    }
}
