class Solution {
    public boolean isValid(String s) 
    {
          Stack<Character> stack = new Stack<Character>();
          boolean flag = false;
          for(int i = 0; i < s.length(); i++)
          {
              char ch = s.charAt(i);
              if(ch == '(' || ch == '[' || ch == '{')
              {
                  stack.push(ch);
              }
              else
              {
                  char c = (stack.size() > 0) ? stack.peek() : ' ';
                  if(ch == ')')
                  {
                      if(stack.size() > 0 && c == '(')
                      {
                          flag = true;
                          stack.pop();
                      }
                      else
                      {
                          flag = false;
                          break;
                      }
                  }
                  if(ch == ']')
                  {
                      if(stack.size() > 0 && c == '[')
                      {
                          flag = true;
                          stack.pop();
                      }
                      else
                      {
                          flag = false;
                          break;
                      }
                  }
                  if(ch == '}')
                  {
                      if(stack.size() > 0 && c == '{')
                      {
                          flag = true;
                          stack.pop();
                      }
                      else
                      {
                          flag = false;
                          break;
                      }
                  }
              }
          }
          return (s.length() == 0) ? true : (stack.size() > 0) ? false : flag;
    }
}
